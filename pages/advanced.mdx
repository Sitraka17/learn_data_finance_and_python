### 4.2 Sharpe Ratio

```python
def sharpe_ratio(returns: pd.Series, risk_free_rate: float = 0.05) -> float:
    excess = returns - risk_free_rate / 252
    return (excess.mean() / excess.std()) * np.sqrt(252)
```

### 4.3 Value at Risk (VaR) & CVaR

```python
def var_historical(returns, confidence=0.95):
    return returns.quantile(1 - confidence)

def cvar(returns, confidence=0.95):
    v = var_historical(returns, confidence)
    return returns[returns <= v].mean()

print(f"VaR  (95%): {var_historical(returns):.4%}")
print(f"CVaR (95%): {cvar(returns):.4%}")
```

### 4.4 Time Value of Money

```python
def npv(rate, cash_flows):
    return sum(cf / (1 + rate)**t for t, cf in enumerate(cash_flows))

def irr(cash_flows):
    from scipy.optimize import brentq
    return brentq(lambda r: npv(r, cash_flows), -0.999, 10)
```

### 5.1 Portfolio Metrics

```python
weights    = np.array([0.25, 0.25, 0.25, 0.25])
cov_matrix = log_returns.cov() * 252

port_return = np.dot(weights, log_returns.mean()) * 252
port_vol    = np.sqrt(weights.T @ cov_matrix @ weights)

print(f"Expected Annual Return : {port_return:.2%}")
print(f"Annual Volatility      : {port_vol:.2%}")
print(f"Sharpe Ratio           : {port_return / port_vol:.2f}")
```

### 5.2 Efficient Frontier (Monte Carlo)

```python
import matplotlib.pyplot as plt

n_portfolios = 10_000
results = np.zeros((3, n_portfolios))

for i in range(n_portfolios):
    w = np.random.dirichlet(np.ones(len(tickers)))
    r = np.dot(w, log_returns.mean()) * 252
    v = np.sqrt(w.T @ cov_matrix @ w)
    results[:, i] = [v, r, r / v]

plt.scatter(results[0], results[1], c=results[2], cmap="viridis", alpha=0.4, s=5)
plt.colorbar(label="Sharpe Ratio")
plt.xlabel("Volatility"); plt.ylabel("Return")
plt.title("Efficient Frontier"); plt.show()
```

### 5.3 Optimal Portfolio (SciPy)

```python
from scipy.optimize import minimize

def neg_sharpe(w, mu, cov):
    return -(np.dot(w, mu)) / np.sqrt(w.T @ cov @ w)

n = len(tickers)
result = minimize(neg_sharpe, np.ones(n)/n,
                  args=(log_returns.mean()*252, cov_matrix),
                  method="SLSQP",
                  bounds=[(0,1)]*n,
                  constraints={"type":"eq","fun": lambda w: w.sum()-1})

print(dict(zip(tickers, result.x.round(4))))
```

### 7.1 Black-Scholes Model

```python
from scipy.stats import norm

def black_scholes(S, K, T, r, sigma, option_type="call"):
    d1 = (np.log(S/K) + (r + 0.5*sigma**2)*T) / (sigma*np.sqrt(T))
    d2 = d1 - sigma*np.sqrt(T)
    if option_type == "call":
        return S*norm.cdf(d1) - K*np.exp(-r*T)*norm.cdf(d2)
    return K*np.exp(-r*T)*norm.cdf(-d2) - S*norm.cdf(-d1)

print(f"Call: ${black_scholes(150, 155, 0.5, 0.05, 0.25):.2f}")
```

### 7.2 The Greeks

```python
def greeks(S, K, T, r, sigma):
    d1 = (np.log(S/K) + (r + 0.5*sigma**2)*T) / (sigma*np.sqrt(T))
    d2 = d1 - sigma*np.sqrt(T)
    return {
        "delta": norm.cdf(d1),
        "gamma": norm.pdf(d1) / (S * sigma * np.sqrt(T)),
        "vega" : S * norm.pdf(d1) * np.sqrt(T) / 100,
        "theta": (-(S*norm.pdf(d1)*sigma)/(2*np.sqrt(T))
                  - r*K*np.exp(-r*T)*norm.cdf(d2)) / 365,
        "rho"  : K*T*np.exp(-r*T)*norm.cdf(d2)/100,
    }
```

### 8.1 Moving Average Crossover

```python
def ma_crossover_strategy(prices, short=50, long=200):
    df = pd.DataFrame({"price": prices})
    df["SMA_short"] = df["price"].rolling(short).mean()
    df["SMA_long"]  = df["price"].rolling(long).mean()
    df["signal"]    = np.where(df["SMA_short"] > df["SMA_long"], 1, -1)
    df["position"]  = df["signal"].shift(1)   # No look-ahead bias
    df["returns"]   = df["price"].pct_change()
    df["strategy"]  = df["position"] * df["returns"]
    return df
```

### 8.2 Performance Metrics

```python
def backtest_metrics(df):
    s = df["strategy"].dropna()
    b = df["returns"].dropna()
    return {
        "Total Return"     : f"{(1+s).prod()-1:.2%}",
        "Benchmark Return" : f"{(1+b).prod()-1:.2%}",
        "Sharpe Ratio"     : f"{s.mean()/s.std()*np.sqrt(252):.2f}",
        "Max Drawdown"     : f"{max_drawdown((1+s).cumprod()):.2%}",
        "Win Rate"         : f"{(s>0).mean():.2%}",
    }
```

### 8.3 RSI Indicator

```python
def rsi(prices, period=14):
    delta = prices.diff()
    gain  = delta.clip(lower=0).rolling(period).mean()
    loss  = (-delta.clip(upper=0)).rolling(period).mean()
    return 100 - (100 / (1 + gain / loss))
```

## Module 9 — Machine Learning in Finance <a name="module-9"></a>

### 🎯 Learning Objectives
- Engineer financial features
- Train ML models for return prediction
- Avoid overfitting and data leakage

### 9.1 Feature Engineering

```python
def build_features(df):
    f = pd.DataFrame(index=df.index)
    f["ret_1d"]    = df["Adj Close"].pct_change(1)
    f["ret_5d"]    = df["Adj Close"].pct_change(5)
    f["ret_20d"]   = df["Adj Close"].pct_change(20)
    f["vol_20d"]   = f["ret_1d"].rolling(20).std()
    f["rsi_14"]    = rsi(df["Adj Close"])
    f["sma_ratio"] = df["Adj Close"] / df["Adj Close"].rolling(50).mean()
    f["target"]    = np.sign(f["ret_1d"].shift(-1))  # Next-day direction
    return f.dropna()
```

### 9.2 Walk-Forward Validation

```python
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import classification_report

features = build_features(aapl)
X = features.drop("target", axis=1)
y = features["target"]

split = int(len(X) * 0.8)                  # Temporal split — never shuffle!
X_train, X_test = X.iloc[:split], X.iloc[split:]
y_train, y_test = y.iloc[:split], y.iloc[split:]

model = RandomForestClassifier(n_estimators=100, max_depth=5, random_state=42)
model.fit(X_train, y_train)
print(classification_report(y_test, model.predict(X_test)))
```

### 9.3 Common Pitfalls

| ❌ Mistake | ✅ Correct Approach |
|-----------|-------------------|
| Random shuffle on time-series | Temporal / walk-forward split |
| Scale on full dataset | Fit scaler only on training set |
| Future data in features | Shift target variable correctly |
| Optimize hyperparameters on test | Use a separate validation window |
